interface{}转其他类型

var a interface{}
	//a = 10
	//a = 10.1
	a = "sss"
	var b int
	var c float64
	var d string
	switch t := a.(type) {
	case int:
		fmt.Println("int")
		b = a.(int)            //interface{} 转 int
		fmt.Println(b)
	case float64:
		c = a.(float64)       //interface{} 转 float
		fmt.Println(c)
	case string:
		d = a.(string)       //interface{} 转 string
		fmt.Println(d)
	default:
		_ = t
		fmt.Println("unKonw")
	}

使用 inter.(type)

inter是interface{}变量的名字，type是具体的类型（int,float32等）
